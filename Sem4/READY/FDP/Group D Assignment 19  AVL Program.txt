node *AVL::insert(node *t,word x)

{

    if(t==NULL)

    {

        t=new node;

        t->data=x;

        t->left=NULL;

        t->right=NULL;



    }

    else

        if(strcmp(x.key,t->data.key)>0)

        {

            t->right=insert(t->right,x);

            if(bf(t)==-2)

                if(strcmp(x.key,t->right->data.key)>0)

                    t=rr(t);

                else

                    t=rl(t);



        }

        else

            if(strcmp(x.key,t->data.key)<0)

        {

            t->left=insert(t->left,x);

            if(bf(t)==2)

                if(strcmp(x.key,t->left->data.key)<0)

                    t=ll(t);

                else

                    t=lr(t);

        }

    return t;

}



node *AVL::rotateright(node *x)

{

    node *y;

    y=x->left;

    x->left=y->right;

    y->right=x;

  

    return(y);

}



node *AVL::rotateleft(node *x)

{

    node *y;

        y=x->right;

        x->right=y->left;

        y->left=x;

     

        return(y);

}



node *AVL::rr(node *t)

{

    t=rotateleft(t);

    return(t);



}



node *AVL::ll(node *t)

{

    t=rotateright(t);

    return(t);



}



node *AVL::lr(node *t)

{

    t->left=rotateleft(t->left);

    t=rotateright(t);

    return(t);

}



node *AVL::rl(node *t)

{

    t->right=rotateright(t->right);

    t=rotateleft(t);

    return(t);

}



int AVL::bf(node *t)

{

    int lh,rh;

    if(t==NULL)

        return 0;

    if(t->left==NULL)

        lh=0;

    else

        lh=1+lh;

    if(t->right==NULL)

        rh=0;

    else

        rh=1+rh;



        return(lh-rh);



}

void AVL::inorder(node *t)

{

    if(t!=NULL)

    {

        inorder(t->left);

        cout<<" ("<<t->data.key<<"  ---  "<<t->data.meaning<<") ";

        inorder(t->right);

    }

}
